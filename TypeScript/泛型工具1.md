#### 泛型工具

泛型工具是一组预定义的泛型类型和操作符，用于操作和转换类型

```ts
//Partial 可选
interface User {
    name:string
    age:number
}

type a = Partial<User> 
//a = User {
//		name?:string
//		age?:number
//}

//原理
type P<T> = {
    [s in keyof T]?:T[s]
}

type a = P<User> 

//Required 必选
interface User {
    name?:string
    age?:number
}

type b = Required<User>
//a = User {
//		name:string
//		age:number
//}

//原理
type R<T> = {
    [s in keyof T]-?:T[s]
}


//Pick
type a = Pick<User,'age'>

//原理
type P<T,k extends keyof T> = {
    [m in k]:T[m]
}

//Exclude

type a = Exclude<'a' | 'b' | 'c','c'>
//原理
type P<T,K> = T extends K ? never : T


//Omit
//用于创建一个新类型，该新类型从原始类型中排除指定的属性
 

type a = Omit<User,'age'>
//原理
type P<T,K extends keyof T> = Pick<T,Exclude<keyof T,K>>  

```
