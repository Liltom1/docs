父子间的传值 ​

传单个字符串不需要v-bind,其它需要v-bind,可简写：

  • 父组件
vue
<template>
  <div>
    <!-- <h2>我是App组件{{ obj.a }}</h2> -->
    <button @click="changeMsg">change</button>
    <Hello title="我是Hello组件" :obj="obj" />
  </div>
</template>
<script setup lang="ts" name="App">
import Hello from './Hello.vue';
//认识 ref,reactive Ref
import type { Ref } from 'vue'
import { ref,reactive, isRef, shallowRef, triggerRef , customRef, readonly , shallowReactive , toRef , toRefs , toRaw} from 'vue' 

const changeMsg = () => {
  obj2.a = 111
}


let obj = reactive<{a:number,data:number[]}>({
  a:1,
  data: [55],
})
const obj2 = reactive(obj)

const obj3 = toRaw(obj2)
console.log(obj3)

</script>
  • 子组件接受
vue
<template>
  <div>
    {{ title }}
  </div>
  <div>
    {{ obj }}
  </div>
  <button @click="onChange">改变</button>
  <button @click="onChange1">改变2</button>
</template>

<script setup lang="ts">
import { computed, reactive, ref , watch ,watchEffect } from 'vue'
//1.第一种方式
// defineProps<{
//   title:string,
//   obj:object
// }>()
//设置默认值
//defineProps({
//    title:{
//        default:"",
//        type:string
//    },
//    data:Array
//})
 
//2.Ts设置默认值的情况
type Props = {
    title?: string,
    obj: {
      a: number,
      data: number[]
    }
}

withDefaults(defineProps<Props>(),{
  title:'hello',

  obj:() => {
    return{
    a:1,
    data:[5]
  }
  } 
})

let p =  reactive({
  nav:{
    bar:{
      name:'hello'
    }
  }
})

let a = ref(1)
let b = ref(8)

function onChange() {
  p.nav.bar.name = 's'
  a.value++
}

function onChange1() {
  b.value++
}
</script>

TIP

请注意，在使用 withDefaults 时，默认值的可变引用类型 (如数组或对象) 应该在函数中进行包装，以避免意外修改和外部副作用。这样可以确保每个组件实例都会获得自己默认值的副本。当使用解构时，这不是必要的。
