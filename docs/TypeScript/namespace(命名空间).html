namespace(命名空间) ​

命名空间中通过export将想要暴露的部分导出

ts
namespace M {
  export const Time: number = 1000
  export const fn = <T>(arg: T): T => {
    return arg
  }
  fn(Time)
}

M.Time
M.fn(1)
  • 嵌套命名空间
ts
//嵌套命名空间
namespace A {
  export namespace B {
    export const Time: number = 1000
    export const fn = <T>(arg: T): T => {
      return arg
    }
    fn(Time)
  }
}
A.B.Time
A.B.fn(1)
